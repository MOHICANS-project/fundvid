48 // Bresenham approach to draw ellipse.
49 // http://raphaello.univ-fcomte.fr/ig/algorithme/ExemplesGLUt/BresenhamEllipse.htm
50 // Add the rotation of the ellipse.
51 // As the algo. use symmetry we must use 4 rotations.
52 template <class Image, class Color>
53 void DrawEllipse(int xc, int yc, int radiusA, int radiusB,
54  const Color &col, Image *pim, double angle = 0.0)
55 {
56  int a = radiusA;
57  int b = radiusB;
58 
59  // Counter Clockwise rotation matrix.
60  double matXY[4] = { cos(angle), sin(angle),
61  -sin(angle), cos(angle)};
62  int x,y;
63  double d1,d2;
64  x = 0;
65  y = b;
66  d1 = b*b - a*a*b + a*a/4;
67 
68  int rotX = (int)ceil(matXY[0] * x + matXY[1] * y);
69  int rotY = (int)ceil(matXY[2] * x + matXY[3] * y);
70  safePutPixel( yc + rotY, xc + rotX, col, pim);
71  rotX = int(matXY[0] * x - matXY[1] * y);
72  rotY = int(matXY[2] * x - matXY[3] * y);
73  safePutPixel( yc + rotY, xc + rotX, col, pim);
74  rotX = int(-matXY[0] * x - matXY[1] * y);
75  rotY = int(-matXY[2] * x - matXY[3] * y);
76  safePutPixel( yc + rotY, xc + rotX, col, pim);
77  rotX = int(-matXY[0] * x + matXY[1] * y);
78  rotY = int(-matXY[2] * x + matXY[3] * y);
79  safePutPixel( yc + rotY, xc + rotX, col, pim);
80 
81  while ( a*a*(y-.5) > b*b*(x+1) ) {
82  if ( d1 < 0 ) {
83  d1 += b*b*(2*x+3);
84  ++x;
85  }
86  else
87  {
88  d1 += b*b*(2*x+3) + a*a*(-2*y+2);
89  ++x;
90  --y;
91  }
92  rotX = int(matXY[0] * x + matXY[1] * y);
93  rotY = int(matXY[2] * x + matXY[3] * y);
94  safePutPixel( yc + rotY, xc + rotX, col, pim);
95  rotX = int(matXY[0] * x - matXY[1] * y);
96  rotY = int(matXY[2] * x - matXY[3] * y);
97  safePutPixel( yc + rotY, xc + rotX, col, pim);
98  rotX = int(-matXY[0] * x - matXY[1] * y);
99  rotY = int(-matXY[2] * x - matXY[3] * y);
100  safePutPixel( yc + rotY, xc + rotX, col, pim);
101  rotX = int(-matXY[0] * x + matXY[1] * y);
102  rotY = int(-matXY[2] * x + matXY[3] * y);
103  safePutPixel( yc + rotY, xc + rotX, col, pim);
104  }
105  d2 = b*b*(x+.5)*(x+.5) + a*a*(y-1)*(y-1) - a*a*b*b;
106  while ( y > 0 ) {
107  if ( d2 < 0 ) {
108  d2 += b*b*(2*x+2) + a*a*(-2*y+3);
109  --y;
110  ++x;
111  }
112  else {
113  d2 += a*a*(-2*y+3);
114  --y;
115  }
116  rotX = int(matXY[0] * x + matXY[1] * y);
117  rotY = int(matXY[2] * x + matXY[3] * y);
118  safePutPixel( yc + rotY, xc + rotX, col, pim);
119  rotX = int(matXY[0] * x - matXY[1] * y);
120  rotY = int(matXY[2] * x - matXY[3] * y);
121  safePutPixel( yc + rotY, xc + rotX, col, pim);
122  rotX = int(-matXY[0] * x - matXY[1] * y);
123  rotY = int(-matXY[2] * x - matXY[3] * y);
124  safePutPixel( yc + rotY, xc + rotX, col, pim);
125  rotX = int(-matXY[0] * x + matXY[1] * y);
126  rotY = int(-matXY[2] * x + matXY[3] * y);
127  safePutPixel( yc + rotY, xc + rotX, col, pim);
128  }
129 }

131 // Bresenham approach do not allow to draw concentric circle without holes.
132 // So it's better the use the Andres method.
133 // http://fr.wikipedia.org/wiki/Algorithme_de_tracé_de_cercle_d'Andres.
134 template <class Image, class Color>
135 void DrawCircle(int x, int y, int radius, const Color &col, Image *pim) {
136  Image &im = *pim;
137  if ( im.Contains(double(y+radius), double(x+radius))
138  || im.Contains(double(y+radius), double(x-radius))
139  || im.Contains(double(y-radius), double(x+radius))
140  || im.Contains(double(y-radius), double(x-radius))) {
141  int x1 = 0;
142  int y1 = radius;
143  int d = radius - 1;
144  while (y1 >= x1) {
145  // Draw the point for each octant.
146  safePutPixel( y1 + y, x1 + x, col, pim);
147  safePutPixel( x1 + y, y1 + x, col, pim);
148  safePutPixel( y1 + y, -x1 + x, col, pim);
149  safePutPixel( x1 + y, -y1 + x, col, pim);
150  safePutPixel(-y1 + y, x1 + x, col, pim);
151  safePutPixel(-x1 + y, y1 + x, col, pim);
152  safePutPixel(-y1 + y, -x1 + x, col, pim);
153  safePutPixel(-x1 + y, -y1 + x, col, pim);
154  if (d >= 2 * x1) {
155  d = d - 2 * x1 - 1;
156  x1 += 1;
157  } else {
158  if (d <= 2 * (radius - y1)) {
159  d = d + 2 * y1 - 1;
160  y1 -= 1;
161  } else {
162  d = d + 2 * (y1 - x1 - 1);
163  y1 -= 1;
164  x1 += 1;
165  }
166  }
167  }
168  }
169 }

172 template <class Image, class Color>
173 void DrawLine(int xa, int ya, int xb, int yb, const Color &col, Image *pim) {
174  Image &im = *pim;
175 
176  // If one point is outside the image
177  // Replace the outside point by the intersection of the line and
178  // the limit (either x=width or y=height).
179  if (!im.Contains(double(ya), double(xa)) || !im.Contains(double(yb), double(xb)))
180  {
181  int width = static_cast<int>(pim->Width());
182  int height = static_cast<int>(pim->Height());
183  const bool xdir = xa<xb, ydir = ya<yb;
184  float nx0 = static_cast<float>(xa), nx1 = static_cast<float>(xb),
185  ny0 = static_cast<float>(ya), ny1 = static_cast<float>(yb),
186  &xleft = xdir?nx0:nx1, &yleft = xdir?ny0:ny1,
187  &xright = xdir?nx1:nx0, &yright = xdir?ny1:ny0,
188  &xup = ydir?nx0:nx1, &yup = ydir?ny0:ny1,
189  &xdown = ydir?nx1:nx0, &ydown = ydir?ny1:ny0;
190 
191  if (xright<0 || xleft>=width) return;
192  if (xleft<0) {
193  yleft -= xleft*(yright - yleft)/(xright - xleft);
194  xleft = 0;
195  }
196  if (xright>=width) {
197  yright -= (xright - width)*(yright - yleft)/(xright - xleft);
198  xright = float(width) - 1;
199  }
200  if (ydown<0 || yup>=height) return;
201  if (yup<0) {
202  xup -= yup*(xdown - xup)/(ydown - yup);
203  yup = 0;
204  }
205  if (ydown>=height) {
206  xdown -= (ydown - height)*(xdown - xup)/(ydown - yup);
207  ydown = float(height) - 1;
208  }
209 
210  xa = (int) xleft;
211  xb = (int) xright;
212  ya = (int) yleft;
213  yb = (int) yright;
214  }
215 
216  int xbas, xhaut, ybas, yhaut;
217  // Check the condition ybas < yhaut.
218  if (ya <= yb) {
219  xbas = xa;
220  ybas = ya;
221  xhaut = xb;
222  yhaut = yb;
223  } else {
224  xbas = xb;
225  ybas = yb;
226  xhaut = xa;
227  yhaut = ya;
228  }
229  // Initialize slope.
230  int x, y, dx, dy, incrmX, incrmY, dp, N, S;
231  dx = xhaut - xbas;
232  dy = yhaut - ybas;
233  if (dx > 0) { // If xhaut > xbas we will increment X.
234  incrmX = 1;
235  } else {
236  incrmX = -1; // else we will decrement X.
237  dx *= -1;
238  }
239  if (dy > 0) { // Positive slope will increment X.
240  incrmY = 1;
241  } else { // Negative slope.
242  incrmY = -1;
243  }
244  if (dx >= dy) {
245  dp = 2 * dy - dx;
246  S = 2 * dy;
247  N = 2 * (dy - dx);
248  y = ybas;
249  x = xbas;
250  while (x != xhaut) {
251  safePutPixel( y, x, col, pim);
252  x += incrmX;
253  if (dp <= 0) { // Go in direction of the South Pixel.
254  dp += S;
255  } else { // Go to the North.
256  dp += N;
257  y+=incrmY;
258  }
259  }
260  } else {
261  dp = 2 * dx - dy;
262  S = 2 * dx;
263  N = 2 * (dx - dy);
264  x = xbas;
265  y = ybas;
266  while (y < yhaut) {
267  safePutPixel( y, x, col, pim);
268  y += incrmY;
269  if (dp <= 0) { // Go in direction of the South Pixel.
270  dp += S;
271  } else { // Go to the North.
272  dp += N;
273  x += incrmX;
274  }
275  }
276  }
277  safePutPixel( y, x, col, pim);
278 }
